# create all tables

set names utf8;

DROP TABLE IF EXISTS MLSSOrganizations;
DROP TABLE IF EXISTS MLSSParticipants;
DROP TABLE IF EXISTS MLSSTalks;
DROP TABLE IF EXISTS MLSSParticipantEmails;
DROP TABLE IF EXISTS MLSSParticipantOrganizations;
DROP TABLE IF EXISTS MLSSTalkSpeakers;
DROP TABLE IF EXISTS MLSSTalkFiles;



CREATE TABLE IF NOT EXISTS MLSSOrganizations (
  id int(10) NOT NULL auto_increment,
  title_ukr VARCHAR(100),
  title_rus VARCHAR(100),
  title_eng VARCHAR(100),
  address VARCHAR(100),
  url VARCHAR(100),
  PRIMARY KEY (id),
  UNIQUE ID (id)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;

CREATE TABLE IF NOT EXISTS MLSSParticipants (
  id int(10) NOT NULL auto_increment,
  name_ukr VARCHAR(50),
  name_rus VARCHAR(50),
  name_eng VARCHAR(50),
  middlename_ukr VARCHAR(50),
  middlename_rus VARCHAR(50),
  middlename_eng VARCHAR(50),
  surname_ukr VARCHAR(50),
  surname_rus VARCHAR(50),
  surname_eng VARCHAR(50),
  sex int(1),
  PRIMARY KEY (id),
  UNIQUE ID (id)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;


CREATE TABLE IF NOT EXISTS MLSSTalks (
  id int(10) NOT NULL auto_increment,
  title_ukr VARCHAR(250),
  title_rus VARCHAR(250),
  title_eng VARCHAR(250),
  date DATETIME,
  org_place_id int(10),
  room int(3),
  PRIMARY KEY (id),
  UNIQUE ID (id)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;


CREATE TABLE IF NOT EXISTS MLSSParticipantEmails (
  part_id int(10),
  email VARCHAR(50)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;



CREATE TABLE IF NOT EXISTS MLSSParticipantOrganizations (
  part_id int(10),
  org_id int(10)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;


CREATE TABLE IF NOT EXISTS MLSSTalkSpeakers (
  talk_id int(10),
  part_id int(10)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;


CREATE TABLE IF NOT EXISTS MLSSTalkFiles (
  talk_id int(10),
  file VARCHAR(100),
  description VARCHAR(300),
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;
